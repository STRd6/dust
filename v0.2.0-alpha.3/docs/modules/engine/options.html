module.exports = (I, self) ->
  optionsMenuShown = false

  $(document).on "keydown", null, "esc", ->
    optionsMenuShown = !optionsMenuShown

    self.pause(optionsMenuShown)

    $(".options").toggleClass "up", optionsMenuShown

    if !optionsMenuShown
      $(".options input").blur()
